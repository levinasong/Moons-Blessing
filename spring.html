  <script>

    const totalFortunes = 28;


    // ❤️ CHANGE THIS LINE ONLY (per flavor)

    const flavorImagePath = "flavor-cards/flavor-01.png";


    const card = document.getElementById("card");

    const button = document.getElementById("revealButton");


    let showingFortune = false;


    window.onload = () => {

      card.src = flavorImagePath;

    };


    function getRandomFortune() {

      const num = Math.floor(Math.random() * totalFortunes) + 1;

      const formatted = String(num).padStart(2, "0");

      return `mobile-cards/mobile-cards-${formatted}.png`;

    }


    button.addEventListener("click", () => {

      if (!showingFortune) {

        card.src = getRandomFortune();

        showingFortune = true;

        button.textContent = "다시 뽑기";

      } else {

        card.src = getRandomFortune();

      }

    });
